Getting Started

# Clone and build git clone https://github.com/nelssec/qualys-mcp cd qualys-mcp go build -o qualys-mcp ./cmd/qualys-mcp # Configure credentials export QUALYS_USERNAME="your-username" export QUALYS_PASSWORD="your-password" export QUALYS_POD="US1" # or EU1, CA1, etc. # Add to Claude Code (claude_desktop_config.json) { "mcpServers": { "qualys": { "command": "/path/to/qualys-mcp" } } }
